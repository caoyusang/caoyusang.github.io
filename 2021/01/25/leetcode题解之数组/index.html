class Solution {
    List<Boolean> res = new ArrayList<>();
    public List<Boolean> kidsWithCandies(int[] candies, int extraCandies) {
        int n = candies.length;
        int maxCandies = Integer.MIN_VALUE;
        for (int i = 0; i < n; i++){
            maxCandies = Math.max(maxCandies, candies[i]);

class Solution {
    public int[] shuffle(int[] nums, int n) {
        int[] res = new int[2 * n];
        for (int i = 0; i < 2 * n; i++){
            if (i % 2 == 1){
                res[i] = nums[n - 1 + (i + 1) / 2];

                res[i] = nums[i / 2];
            }
        }
        return res;
    }

1512.“好对儿”的数目
题解
class Solution {
    HashMap<Integer, Integer> map = new HashMap<>();
    public int numIdenticalPairs(int[] nums) {
        int n = nums.length;
        int cnt = 0;
        for (int i = 0; i < n; i++){
            map.put(nums[i], map.getOrDefault(nums[i], 0) + 1);
        }
        
        for (int num : map.keySet()){
            if (map.containsKey(num)){
                int val = map.get(num);
                //该数有"好对儿"
                if (val > 1){
                    cnt += (val - 1) * val / 2;
                }
            }
        }
        return cnt;
        
    }
}

class Solution {
    public int sumOddLengthSubarrays(int[] arr) {
        int sum = 0;
        int n = arr.length;
        //k表示子数组的长度，为奇数
        for (int k = 1; k <= n; k += 2){
            for (int i = 0; i + k  <= n; i++){
                int tmp = arr[i];
                for (int j = 1; j < k; j++){
                     tmp += arr[i + j];
                }
                sum += tmp;
            }
        }
        return sum;
    }
}
1732.找到最高海拔

class Solution {
    public int largestAltitude(int[] gain) {
        int maxH = Integer.MIN_VALUE;
        int curH = 0;
        for (int i = 0; i < gain.length; i++){
            maxH = Math.max(maxH, curH);
            curH += gain[i];
        }
        //记得最后再求一次最大，因为curH最后一次赋值后未被比较
        maxH = Math.max(maxH, curH);
        return maxH;
    }
}

class Solution {
    public int[][] flipAndInvertImage(int[][] A) {
        int m = A.length, n = A[0].length;
        for(int[] arr : A){
            //翻转的同时进行异或(0 ^ 1 = 1)
            for (int i = 0; i < (n + 1) / 2; i++){
                int tmp = arr[n - i - 1] ^ 1;

            }
        }
        
        return A;
    }
}
1572.矩阵的对角线之和
题解
class Solution {
    public int diagonalSum(int[][] mat) {
        int n = mat.length;
        int sum = 0;
        for (int i = 0; i < n; i++){
            sum += mat[i][i];
        }
        for (int i = n - 1; i >= 0; i--){
            sum += mat[i][n - i - 1];
        }
        //偶阶方阵不会有主副对角线元素重复，奇阶方阵需要减去重复的
        return sum = n % 2 == 0 ? sum : sum - mat[n / 2][n / 2];
    }
}

    public int maxProduct(int[] nums) {
        int max1 = nums[0], max2 = nums[1];
        if(max1 < max2){
            int t = max1;
            max1 = max2;
            max2 = t;
        }
        for(int i = 2; i < nums.length; i++){
            if(nums[i] > max1){
                max2 = max1;
                max1 = nums[i];
            }else if(nums[i] > max2){
                max2 = nums[i];
            }
        }
        return (max1-1)*(max2-1);
    }
}

1351.在有序矩阵中统计负数的总个数

矩阵的行列都为降序

题解
class Solution {
    public int countNegatives(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        int firstNegativeIdx = n;
        int count = 0;
        for (int i = 0; i < m; i++){
            firstNegativeIdx = findFirstNegativeIdx(grid[i], firstNegativeIdx);
            count += grid[i].length - firstNegativeIdx;
        }
        return count;
    }
    
    
    public int findFirstNegativeIdx(int[] arr, int limit) {
        int res = limit;
        int lo = 0;
        int hi = limit;
        while (lo < hi) {
            int mid = lo + (hi - lo) / 2;
            if (arr[mid] < 0) {
                res = mid;
                hi = mid;
            } else {
                lo = mid + 1;
            }
        }
        return res;
    }
}

48.顺时针90°旋转矩阵

要求就地修改，即O(1)

题解
class Solution {
    public void rotate(int[][] matrix) {
        int n = matrix.length;
        for (int i = 0; i < (n + 1) / 2; i++) {
            for (int j = 0; j < n / 2; j++) {
                //顺时针交换值
                int temp = matrix[n - 1 - j][i];
                matrix[n - 1 - j][i] = matrix[n - 1 - i][n - j - 1];
                matrix[n - 1 - i][n - j - 1] = matrix[j][n - 1 - i];
                matrix[j][n - 1 - i] = matrix[i][j];
                matrix[i][j] = temp;
            }
        }
    }
}

class Solution {
    public void moveZeroes(int[] nums) {
        int n = nums.length;
        //用两个计数器分别计数非零元素和赋值下标
        int zeroCnt = 0, nonZeroCnt = 0;
        for (int k = 0; k < n; k++){
            if (nums[k] == 0){
                zeroCnt++;

                nums[nonZeroCnt++] = nums[k];

        }

78.数组的子集
题解
class Solution {
    List<List<Integer>> res = new ArrayList<>();
    int n, k;
    public List<List<Integer>> subsets(int[] nums) {
        n = nums.length;
        //k为子数组长度，每次都从0开始，增长到k就停止
        for (k = 0; k < n + 1; k++){
            backTrack(new ArrayList<Integer>(), nums, 0);
        }
        return res;
    }
    
    public void backTrack(ArrayList<Integer> cur, int[] nums, int size){
        //子数组长度增长到k，结束
        if (cur.size() == k){
            res.add(new ArrayList(cur));
            return;
        }
        for (int i = size; i < n; i++){
            cur.add(nums[i]);
            //继续组合
            backTrack(cur, nums, i + 1);
            //回溯
            cur.remove(cur.size() - 1);
        }
    }
}
238.构建乘积数组
题解
class Solution {
    public int[] productExceptSelf(int[] nums) {
        int n = nums.length;
        int[] res = new int[n];
        Arrays.fill(res, 1);
        //左半部分元素累乘(第一个元素没有左半部分)
        for (int i = 1; i < n; i++){
            res[i] = res[i - 1] * nums[i - 1];
        }
        int tmp = 1;
        //右半部分元素(最后一个元素没有右半部分)
        for (int j = n - 2; j >= 0; j--){
            tmp *= nums[j + 1];
            //乘上右半部分的累乘
            res[j] *= tmp;
        }
        return res;
        
        
    }
}

64.矩阵中的最小路径和
题解
class Solution {
    public int minPathSum(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        //dp[i][j]代表(0,0)到(i,j)的最短路径
        int[][] dp = new int[m][n];
        dp[0][0] = grid[0][0];
        for (int i = 1; i < m; i++){
            dp[i][0] = dp[i - 1][0] + grid[i][0];
        }
        for (int i = 1; i < n; i++){
            dp[0][i] = dp[0][i - 1] + grid[0][i];
        }
        for (int i = 1; i < m; i++){
            for (int j = 1; j < n; j++){
                dp[i][j] = Math.min(dp[i - 1][j], dp[i][j - 1]) + grid[i][j];
            }
        }
        return dp[m - 1][n - 1];
    }
}

200.岛屿数目
题解
class Solution {
    public int numIslands(char[][] grid) {
        int m = grid.length, n = grid[0].length;
        int islands = 0;
        for (int i = 0; i < m; i++){
            for (int j = 0; j < n; j++){
                //遍历到'1'，说明是岛屿
                if (grid[i][j] == '1'){
                    islands++;
                    //深度优先，将与这个岛相连的陆地都变成'0'，那么下次再碰到一个'1'时就是找到一个新的岛
                    dfs(grid, i, j);
                }
            }
        }
        return islands;
    }
    
    public void dfs(char[][] arr, int i, int j){
        //已经到岛屿的边缘，则停止搜索
        if (arr[i][j] == '0') return;
        arr[i][j] = '0';
        if (i - 1 >= 0) dfs(arr, i - 1, j);
        if (i + 1 <= arr.length - 1) dfs(arr, i + 1, j);
        if (j - 1 >= 0) dfs(arr, i, j - 1);
        if (j + 1 <= arr[0].length - 1) dfs(arr, i, j + 1);
    }
}
118.帕斯卡三角形
题解
class Solution {
    List<List<Integer>> res = new ArrayList<>();
    public List<List<Integer>> generate(int numRows) {
        if (numRows == 0) return res;
        //第一行
        res.add(new ArrayList<Integer>());
        res.get(0).add(1);
        //第二行往后
        for (int i = 1; i < numRows; i++){
            List<Integer> tmp = new ArrayList<>();
            //前一行
            List<Integer> pre = res.get(i - 1);
            //List第一个元素
            tmp.add(1);
            for (int j = 1; j < i; j++){
                tmp.add(pre.get(j - 1) + pre.get(j));
            }
            //List最后一个元素
            tmp.add(1);
            res.add(tmp);
        }
        return res;
    }
}
350.两个数组的交集Ⅱ

class Solution {
    public int[] intersect(int[] nums1, int[] nums2) {
        List<Integer> res = new ArrayList<>();
        Arrays.sort(nums1);
        Arrays.sort(nums2);

        int p = 0, q = 0;
        while (p < nums1.length && q < nums2.length){
            if (nums1[p] == nums2[q]){
                res.add(nums1[p]);
                p++;
                q++;
            } else if (nums1[p] < nums2[q]){
                p++;
            } else {
                q++;
            }
        }
        
        int n = res.size();
        int[] arr = new int[n];
        for (int i = 0; i < n; i++){
            arr[i] = res.get(i);
        }
        return arr;
    }
}
39.组合和
题解
class Solution {
    List<List<Integer>> res = new ArrayList<>();
    public List<List<Integer>> combinationSum(int[] candidates, int target) {
        if(candidates==null || candidates.length==0) return res;
        // Arrays.sort(candidates);
        backTrack(candidates, new ArrayList<Integer>(), 0, 0, target);
        return res;
    }
    
    public void backTrack(int[] nums, ArrayList<Integer> lst, int index, int cur, int k){
        //组合和大于等于target时，终止
        if (cur >= k){
            //等于时相当于找到一个组合和
            if (cur == k){
                res.add(new ArrayList<>(lst));
            }
            return;
        }
        //index防止出现重复情况
        for (int i = index; i < nums.length; i++){
            if (cur <= k - nums[i]){
                lst.add(nums[i]);
                backTrack(nums, lst, i, cur, k - nums[i]);
                lst.remove(lst.size() - 1);
            }
        }
    }
}

55.跳跃游戏
题解

贪心

class Solution {
    public boolean canJump(int[] nums) {
        int n = nums.length;
        int farthest = 0;
        //当所遍历的位置还没超出可farthest范围时，根据跳力更新可reach范围
        for (int i = 0; i <= farthest; i++){
            //通过每一步能跳到的最远位置，更新全局能跳到的最远位置
            farthest = Math.max(farthest, i + nums[i]);
            if (farthest >= n - 1) return true;
        }
        return false;
    }
}
45.跳跃游戏Ⅱ

保证能跳到最后一个位置，那么至少需要跳几次？

题解
class Solution {
    public int jump(int[] nums) {
        int n = nums.length;
        int reach = 0, nextReach = 0;
        int jumps = 0;
        for (int i = 0; i < n - 1; i++){
            nextReach = Math.max(nextReach, i + nums[i]);
            if (reach == i){
                jumps++;
                reach = nextReach;
            }
        }
        return jumps;
    }
}
581.最短的无序连续子序列的长度

给定一个整数数组nums，你需要找到一个连续的子数组，如果你只排序这个子数组升序，那么整个数组将升序排序。

返回最短的子数组并输出其长度。

题解
class Solution {
    public int findUnsortedSubarray(int[] nums) {
        int n = nums.length;
        if (n == 1) return 0;
        //克隆一个nums
        int[] cnums = nums.clone();
        int left = Integer.MAX_VALUE, right = Integer.MIN_VALUE;
        //对nums排序
        Arrays.sort(nums);
        for (int i = 0; i < n; i++){
            if (cnums[i] != nums[i]){
                left = Math.min(left, i);
                right = Math.max(right, i);
            }
        }
        
        return right > left ? right - left + 1 : 0;
    }
}
