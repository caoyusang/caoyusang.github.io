手撕排序算法

冒泡排序

时间复杂度O(n*n)，空间复杂度O(n)，稳定排序

//冒泡排序——每次把数组无序部分的最大值交换到数组最末尾
public static void bubbleSort(int[] nums){
    int len = nums.length;
    //记录上一次最后完成交换的位置，因为后面的已经有序了，这样可以避免无意义的比较判断
    int lastSwap = len;
    //外循环——冒泡交换的趟数
    for (int i = 0; i < len; i++){
        //用于判断该趟有无元素交换，没有就说明数组已经有序了，可以break跳出循环
        boolean hasSwap = false;
        //暂存需要比较到的位置
        int sortBorder = lastSwap;
        for (int j = 1; j < sortBorder; j++){
            if (nums[j] < nums[j - 1]){
                //进行元素交换
                int temp = nums[j];
                nums[j] = nums[j - 1];
                nums[j - 1] = temp;
                //发生交换
                hasSwap = true;
                //更新最后交换元素的位置
                lastSwap = j;
            }
        }
        //如果该趟没有发生元素交换，直接跳出循环
        if (!hasSwap) break;
    }
}

选择排序

时间复杂度O(n*n)，空间复杂度O(n)，不稳定排序

//选择排序——找出无序部分中的最小值，放到有序部分的末尾
public static void selectSort(int[] nums){
    int n = nums.length;
    for (int i = 0; i < n; i++){
        int min = i;
        for (int j = i + 1; j < n; j++){
            //找当前数组无序部分的最小值的下标，将对应的数放到有序部分末尾
            if (nums[j] < nums[min]){
                min = j;
            }
        }
        //做交换
        int temp = nums[min];
        nums[min] = nums[i];
        nums[i] = temp;
    }
}

插入排序

时间复杂度O(n*n)，空间复杂度O(1)，稳定排序

//插入排序，将数组无序部分的元素向前插入到数组有序部分的对应位置
public static void insertSort(int[] nums){
    int n = nums.length;
    //第一个元素无法向前插入
    for (int i = 1; i < n; i++){
        //待插入元素
        int val = nums[i];
        int j = 0;
        for (j = i - 1; j >= 0; j--){
            //待插入元素比前一个元素小，则让前一个元素覆盖它，继续向前查找插入位置，找到就break
            if (val < nums[j]){
                nums[j + 1] = nums[j];
            } else {
                break;
            }
        }
        //插入元素
        nums[j + 1] = val;
    }
}

快速排序

时间复杂度O(n*log(n))，空间复杂度O(log(n))，不稳定排序

//递归快排
public static void quickSort(int start, int end, int[] nums){
    if (start >= end) return;
    int border = partion(start, end, nums);
    quickSort(start, border - 1, nums);
    quickSort(border + 1, end, nums);
}

//快排——双边扫描，把比基准元素放到正确的位置，基准元素左边全是比它小的，右边全是比它大的
public static int partion(int start, int end, int[] nums){
    int left = start, right = end;
    //选取当前分区第一个元素作为基准元素(可以有别的选取基准元素的方法)
    int pivot = nums[start];
    while (left != right){
        //右哨兵right向左搜索第一个比pivot小的元素
        while(nums[right] >= pivot && left < right) right--;
        //左哨兵left向右搜索第一个比pivot大的元素
        while(nums[left] <= pivot && left < right) left++;

        //如果一轮搜索完毕后，left < right，将nums[left]与nums[right]互换，使得小的元素在前，大的元素在后
        if (left < right){
            swap(nums, left, right);
        }
    }
    //将基准元素放到对的位置，即此时将nums[start]和nums[left]互换
    nums[start] = nums[left];
    nums[left] = pivot;
    //返回基准元素的位置，作为分区的界限
    return left;
}

//用于数组的元素交换
public static void swap(int[] arr, int i, int j){
    int temp = arr[i];
    arr[i] = arr[j];
    arr[j] = temp;
}

归并排序

时间复杂度O(n*log(n))，空间复杂度O(n)，稳定排序

//分治
public static void divideMerge(int[] nums, int left, int right, int[] temp){
    if (left < right){
        int mid = (left + right) / 2;
        //递归
        divideMerge(nums, left, mid, temp);
        divideMerge(nums, mid + 1, right, temp);
        //合并结果
        merge(nums, left, mid, right, temp);
    }
}
    
//merge
public static void merge(int[] nums, int left, int mid, int right, int[] temp){
    //左半部分的数组指针
    int i = left;
    //右半部分的数组指针
    int j = mid + 1;
    //合并的临时数组指针
    int t = 0;
    
    //合并过程
    while (i <= mid && j <= right){
        if (nums[i] <= nums[j]){
            temp[t++] = nums[i++];
        } else {
            temp[t++] = nums[j++];
        }
    }
    
    //比较合并完如果还有剩余，则将剩余的元素直接进行填充
    while (i <= mid){
        temp[t++] = nums[i++];
    }
    while (j <= right){
        temp[t++] = nums[j++];
    }
    
    //将临时数组拷贝到原始数组中，先将临时数组的指针置零
    t = 0;
    while (left <= right){
        nums[left++] = temp[t++];
    }
}

堆排序

时间复杂度O(n*log(n))，空间复杂度O(1)，不稳定排序

将待排序序列构造成一个大顶堆，此时，整个序列的最大值就是堆顶的根节点。将其与末尾元素进行交换，此时末尾就为最大值。然后将剩余n-1个元素重新构造成一个堆，这样会得到n个元素的次小值。如此反复执行，便能得到一个有序序列。

public static void heapSort(int[] nums){
    //从第一个非叶子节点(0 ~ nums.length - 1编号)开始，自下而上，自右向左调整结构，使之成为最大堆
    for (int i = nums.length / 2 - 1; i >= 0; i--){
        adjustHeap(nums, i, nums.length);
    }
    //将堆顶元素和数组末尾元素进行交换，然后针对剩下的n-1个元素再次进行堆调整
    for (int j = nums.length - 1; j >= 0; j--){
        swap(nums, 0, j);
        adjustHeap(nums, 0, j);
    }
}

public static void adjustHeap(int[] nums, int i, int len){
    //暂存当前元素
    int temp = nums[i];
    for (int k = 2 * i + 1; k < len; k = k * 2 + 1){
        //当前节点的左子节点的值小于右子节点
        if (k < len - 1 && nums[k] < nums[k + 1]){
            k++;
        }
        //子节点比父节点大，把子节点的值赋值给父节点就行，因为已经对当前父节点进行了暂存
        if (nums[k] > temp){
            nums[i] = nums[k];
            i = k;
        } else {
            break;
        }
        
    }
    //把之前的父节点放到调整后的位置上
    nums[i] = temp;
}

//数组元素交换
public static void swap(int[] arr, int a, int b){
    int temp = arr[a];
    arr[a] = arr[b];
    arr[b] = temp;
}
